95 struct cache_param
96 {
97  cache_param(scm_file *file) : n(int(file->get_w()) - 2),
98  c(int(file->get_c())),
99  b(int(file->get_b())) { }
100 
101  int n; // Page size
102  int c; // Channels per pixel
103  int b; // Bits per channel
104 
105  bool operator<(const cache_param& that) const {
106  if (n < that.n) return true;
107  else if (n > that.n) return false;
108  else if (c < that.c) return true;
109  else if (c > that.c) return false;
110  else if (b < that.b) return true;
111  else return false;
112  }
113 };
