83 template <typename T> bool scm_queue<T>::try_insert(T& d)
84 {
85  if (SDL_SemTryWait(free_slots) == 0)
86  {
87  SDL_LockMutex(data_mutex);
88  {
89  S.insert(d);

91  SDL_UnlockMutex(data_mutex);
